#include <avr/io.h>
#include <util/delay.h>

/*
 * Assumptions:
 * 	- LED connected to PORTB.5
 * 	- Switch connected to PORTD.2
 */

int main (void)
{
	/* set PORTB for output*/
	DDRB = 0xFF;
	/* set PORTD for input*/
	DDRD &= 0xFB;
	PORTD |= 0x04;

	while (1) {
		if (PIND & 0x04)
			PORTB &= ~0x20;
		else
			PORTB |= 0x20;
	}
	return 0;
}
